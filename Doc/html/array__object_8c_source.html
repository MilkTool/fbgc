3 size_t array_calculate_new_capacity_from_size(size_t size){
4  /*
5  Below algorithm calculates the capacity for the given size
6  Basically capacity is the closest two's power
7  0,1,2 : 2
8  3,4 : 4
9  5,6,7,8 : 8
10  9,10,11,12,13,14,15 : 16
11  and so on
12 
13  Take 5 for example, in binary its 0b00101
14  take z = 1
15  0b0001 <= 0b0101 , shift z left
16  0b0010 <= 0b0101 , (ditto)
17  0b0100 <= 0b0101 , (ditto)
18  0b1000 <= 0b0101 , stop here don't shift, z is 8, the closest two's power for 5
19  */
20  size_t z = 1;
21  while(z < size)
22  z <<= 1;
23 
24  #ifdef ARRAY_DEBUG
25  cprintf(011,"For the size %d new capacity is calculated as :%d\n",size,z);
26  #endif
27  return z;
